    361                                                  virtual protected task get_and_drive();

    363                                                          @(posedge axiLiteVif.ACLK);
    364                                                          if (axiLiteVif.ARESETN == 1'b0) begin
    365                                                              @(posedge axiLiteVif.ARESETN);

    367                                                          end
    368                                                          seq_item_port.get_next_item(req);

    371                                                              @(posedge axiLiteVif.ACLK);

    433                                                      axiLiteVif.ARPROT  <= 3'h0;
    434                                                      axiLiteVif.ARVALID <= 1'b1;
    435                                                      for(axiLite_ctr = 0; axiLite_ctr <= 31; axiLite_ctr ++) begin

    448                                                      int axiLite_ctr;

    450                                                      axiLiteVif.WSTRB  <= 4'hf;

    496                                                          `uvm_error("axiLite_master_driver","Received ERROR Read Response");
    497                                                          axiLiteVif.RREADY <= axiLiteVif.RVALID;
    498                                                          @(posedge axiLiteVif.ACLK);
    499                                                      end

    517                                                  virtual task drive();
    518                                                      template_transaction tx;
    519                                                      integer counter = 0, state = 0;
    520                                                      templateVif.sig_ina = 0'b0;
    521                                                      templateVif.sig_inb = 0'b0;
    522                                                      templateVif.sig_en_i = 1'b0;
    523                                                      forever begin
    524                                                          if(counter==0)
    525                                                          begin
    526                                                              seq_item_port.get_next_item(tx);
    527                                                          end
    528                                                          @(posedge templateVif.sig_clock)
    529                                                          begin
    530                                                              if(counter==0)
    531                                                              begin
    532                                                                  templateVif.sig_en_i = 1'b1;
    533                                                                  state = 1;
    534                                                              end
    535                                                              if(counter==1)
    536                                                              begin
    537                                                                  templateVif.sig_en_i = 1'b0;
    538                                                              end
    539                                                              case(state)
    540                                                                  1: begin
    541                                                                      templateVif.sig_ina = tx.ina[1];
    542                                                                      templateVif.sig_inb = tx.inb[1];
    543                                                                      tx.ina = tx.ina << 1;
    544                                                                      tx.inb = tx.inb << 1;
    545                                                                      counter = counter + 1;
    546                                                                      if(counter==2) state = 2;
    547                                                                  end
    548                                                                  2: begin
    549                                                                      templateVif.sig_ina = 1'b0;
    550                                                                      templateVif.sig_inb = 1'b0;
    551                                                                      counter = counter + 1;
    552                                                                      if(counter==6)
    553                                                                      begin
    554                                                                          counter = 0;
    555                                                                          state = 0;
    556                                                                          seq_item_port.item_done();
    557                                                                      end
    558                                                                  end
    559                                                              endcase
    560                                                          end
    561                                                      end
    562                                                  endtask: drive
    563                                              endclass: template_driver
